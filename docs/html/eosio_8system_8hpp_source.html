47 template<typename E, typename F>
48 static inline auto has_field( F flags, E field )
49 -> std::enable_if_t< std::is_integral_v<F> && std::is_unsigned_v<F> &&
50 std::is_enum_v<E> && std::is_same_v< F, std::underlying_type_t<E> >, bool>
51 {
52 return ( (flags & static_cast<F>(field)) != 0 );
53 }
54
55 template<typename E, typename F>
56 static inline auto set_field( F flags, E field, bool value = true )
57 -> std::enable_if_t< std::is_integral_v<F> && std::is_unsigned_v<F> &&
58 std::is_enum_v<E> && std::is_same_v< F, std::underlying_type_t<E> >, F >
59 {
60 if( value )
61 return ( flags | static_cast<F>(field) );
62 else
63 return ( flags & ~static_cast<F>(field) );
64 }

235 // The unregprod and claimrewards actions modify unrelated fields of the producers table and under the default
236 // serialization behavior they would increase the size of the serialized table if the producer_authority field
237 // was not already present. This is acceptable (though not necessarily desired) because those two actions require
238 // the authority of the producer who pays for the table rows.
239 // However, the rmvproducer action and the onblock transaction would also modify the producer table in a similar
240 // way and increasing its serialized size is not acceptable in that context.
241 // So, a custom serialization is defined to handle the binary_extension producer_authority
242 // field in the desired way. (Note: v1.9.0 did not have this custom serialization behavior.)
243
244 template<typename DataStream>
245 friend DataStream& operator << ( DataStream& ds, const producer_info& t ) {
246 ds << t.owner
247 << t.total_votes
248 << t.producer_key
249 << t.is_active
250 << t.url
251 << t.unpaid_blocks
252 << t.last_claim_time
253 << t.location;
254
255 if( !t.producer_authority.has_value() ) return ds;
256
257 return ds << t.producer_authority;
258 }

261 friend DataStream& operator >> ( DataStream& ds, producer_info& t ) {

553 // to gradually phase out staking and REX, 0.01x (10^13) is a good
554 // value for both current_weight_ratio and target_weight_ratio.
555 std::optional<int64_t> target_weight_ratio; // Linearly shrink weight_ratio to this amount. 1x = 10^15. 0.01x = 10^13.
556 // Do not specify to preserve the existing setting or use the default.
557 std::optional<int64_t> assumed_stake_weight; // Assumed stake weight for ratio calculations. Use the sum of total
558 // staked and total rented by REX at the time the power market
559 // is first activated. Do not specify to preserve the existing
560 // setting (no default exists); this avoids sudden price jumps.
561 // For new chains which don't need to phase out staking and REX,
562 // 10^12 is probably a good value.
563 std::optional<time_point_sec> target_timestamp; // Stop automatic weight_ratio shrinkage at this time. Once this
564 // time hits, weight_ratio will be target_weight_ratio. Ignored
565 // if current_weight_ratio == target_weight_ratio. Do not specify
566 // this to preserve the existing setting (no default exists).
567 std::optional<double> exponent; // Exponent of resource price curve. Must be >= 1. Do not specify
568 // to preserve the existing setting or use the default.
569 std::optional<uint32_t> decay_secs; // Number of seconds for the gap between adjusted resource
570 // utilization and instantaneous resource utilization to shrink
571 // by 63%. Do not specify to preserve the existing setting or
572 // use the default.
573 std::optional<asset> min_price; // Fee needed to reserve the entire resource market weight at the
574 // minimum price. For example, this could be set to 0.005% of
575 // total token supply. Do not specify to preserve the existing
576 // setting or use the default.
577 std::optional<asset> max_price; // Fee needed to reserve the entire resource market weight at the
578 // maximum price. For example, this could be set to 10% of total
579 // token supply. Do not specify to preserve the existing
580 // setting (no default exists).
581
582 EOSLIB_SERIALIZE( powerup_config_resource, (current_weight_ratio)(target_weight_ratio)(assumed_stake_weight)
583 (target_timestamp)(exponent)(decay_secs)(min_price)(max_price) )
584 };

597 struct powerup_state_resource {
598 static constexpr double default_exponent = 2.0; // Exponent of 2.0 means that the price to reserve a
599 // tiny amount of resources increases linearly
600 // with utilization.
601 static constexpr uint32_t default_decay_secs = 1 * seconds_per_day; // 1 day; if 100% of bandwidth resources are in a
602 // single loan, then, assuming no further powerup usage,
603 // 1 day after it expires the adjusted utilization
604 // will be at approximately 37% and after 3 days
605 // the adjusted utilization will be less than 5%.
606
607 uint8_t version = 0;
608 int64_t weight = 0; // resource market weight. calculated; varies over time.
609 // 1 represents the same amount of resources as 1
610 // satoshi of SYS staked.
611 int64_t weight_ratio = 0; // resource market weight ratio:
612 // assumed_stake_weight / (assumed_stake_weight + weight).
613 // calculated; varies over time. 1x = 10^15. 0.01x = 10^13.
614 int64_t assumed_stake_weight = 0; // Assumed stake weight for ratio calculations.
615 int64_t initial_weight_ratio = powerup_frac; // Initial weight_ratio used for linear shrinkage.
616 int64_t target_weight_ratio = powerup_frac / 100; // Linearly shrink the weight_ratio to this amount.
617 time_point_sec initial_timestamp = {}; // When weight_ratio shrinkage started
618 time_point_sec target_timestamp = {}; // Stop automatic weight_ratio shrinkage at this time. Once this
619 // time hits, weight_ratio will be target_weight_ratio.
620 double exponent = default_exponent; // Exponent of resource price curve.
621 uint32_t decay_secs = default_decay_secs; // Number of seconds for the gap between adjusted resource
622 // utilization and instantaneous utilization to shrink by 63%.
623 asset min_price = {}; // Fee needed to reserve the entire resource market weight at
624 // the minimum price (defaults to 0).
625 asset max_price = {}; // Fee needed to reserve the entire resource market weight at
626 // the maximum price.
627 int64_t utilization = 0; // Instantaneous resource utilization. This is the current
628 // amount sold. utilization <= weight.
629 int64_t adjusted_utilization = 0; // Adjusted resource utilization. This is >= utilization and
630 // <= weight. It grows instantly but decays exponentially.
631 time_point_sec utilization_timestamp = {}; // When adjusted_utilization was last updated
